    <form action="https://www.google.com/search" method="get" class="search-box" id="searchForm">

        <input type="text" name="q" id="query" placeholder="Enter file name or keywords">

        <select name="filetype" id="filetype">

            <option value="pdf">PDF</option>

            <option value="doc">Word Documents</option>

            <option value="ppt">PowerPoint</option>

            <option value="xls">Excel</option>

            <option value="txt">Text Files</option>

        </select>

        <input type="submit" value="Search Files">

    </form>

    <script>

        // Get form elements

        const form = document.getElementById('searchForm');

        const queryInput = document.getElementById('query');

        const fileTypeSelect = document.getElementById('filetype');

        // Clear text when input is focused

        queryInput.addEventListener('focus', function() {

            this.value = '';

        });

        // Update query value with keywords and file type

        form.addEventListener('submit', function(event) {

            event.preventDefault();  // Prevent the form from submitting immediately

            const keywords = queryInput.value.trim();

            const filetype = fileTypeSelect.value;

            queryInput.value = `${keywords} filetype:${filetype}`;

            

            // Submit the form after updating the value

            form.submit();

        });

    </script>
